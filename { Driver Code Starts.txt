#include <map>
class Solution{
public:
    int LargButMinFreq(int arr[], int n) {
        // code here
        map<int,int>ans;
        for(int i = 0 ; i < n; i++)
        {
            ans[arr[i]]++;
        }
        int min = INT_MAX;
        for(auto i: ans){
            if(i.second < min)
               min = i.second;
        }
        int max = INT_MIN;
        for(auto i: ans){
            if(i.second == min)              
              if( max < i.first)
                  max = i.first;
        }
        return max;
        
        
    }
};
